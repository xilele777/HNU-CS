# 2.程序填空题（15 分，每空 3 分）

如下是一个 C 语言程序及其对应的汇编代码，请参照汇编代码，完成 C 程序的空缺部分。

## C 语言程序：

```c
#include <stdio.h>
int main()
{
    int i,j,k=0,loop;
    int* p;
    int arr[3][2]={{9,5},{ (1) ,22},{6,18}};
    scanf("%d,%d",&i,&j);
    p=arr+2;
    do
    {
        if( (2) )
            k+=*p;
        else
            k+= (3) ;
    }while( (4) );
    return (5) ;
}
```

## 答案与解析

### (1) `31`

汇编中 `movl $31, 32(%esp)` 对应 `arr[1][0]` 的初始化值。

栈布局（从 24(%esp) 开始）：
- 24(%esp) = arr[0][0] = 9
- 28(%esp) = arr[0][1] = 5
- **32(%esp) = arr[1][0] = 31**
- 36(%esp) = arr[1][1] = 22
- 40(%esp) = arr[2][0] = 6
- 44(%esp) = arr[2][1] = 18

---

### (2) `i*j - *p > 6`

汇编对应段：

```asm
.L6:
    movl 48(%esp), %edx       # edx = i
    movl 52(%esp), %eax       # eax = j
    imull %eax, %edx          # edx = i * j
    movl 60(%esp), %eax       # eax = p
    movl (%eax), %eax         # eax = *p
    movl %edx, %ecx           # ecx = i * j
    subl %eax, %ecx           # ecx = i*j - *p
    cmpl $6, %eax             # (实际为 cmpl $6, %ecx，因上条 subl 结果在 ecx→eax)
    jle .L2                   # 若 i*j - *p <= 6，跳转到 .L2
    # 否则（i*j - *p > 6）：
    ...                       # k += *p
```

`jle .L2` 表示 **≤6 时跳转**，即不跳转（执行 `k+=*p`）的条件是 **`i*j - *p > 6`**。

---

### (3) `*(p-4)>2 ? k-2*i : i+j`

汇编对应段（else 分支，即 .L2 标签处）：

```asm
.L2:
    movl 60(%esp), %eax       # eax = p
    subl $4, %eax             # eax = p - 4（地址减 4 字节 = 前一个 int）
    movl (%eax), %eax         # eax = *(p-4) = arr[1][1] = 22
    cmpl $2, %eax             # 比较 *(p-4) 与 2
    jg .L4                    # 若 *(p-4) > 2，跳转到 .L4

    # *(p-4) ≤ 2 时（注意：实际 *(p-4)=22，此分支恒不执行）：
    movl 48(%esp), %edx       # edx = i
    movl 52(%esp), %eax       # eax = j
    addl %edx, %eax           # eax = i + j
    jmp .L5

.L4:
    # *(p-4) > 2 时：
    movl 48(%esp), %edx       # edx = i
    movl $0, %eax             # eax = 0
    subl %edx, %eax           # eax = -i
    addl %eax, %eax           # eax = -2i
    addl 56(%esp), %eax       # eax = k - 2i

.L5:
    addl %eax, 56(%esp)       # k += eax
```

两种情况的语义可合并为一个三元表达式：`*(p-4)>2 ? k-2*i : i+j`。

---

### (4) `i++ * j-- > 0 || i <= 4`

汇编对应段（.L3 标签 — 循环条件判断与迭代变量的更新）：

```asm
.L3:
    movl 48(%esp), %edx       # edx = i
    movl 52(%esp), %eax       # eax = j
    movl %edx, %ecx           # ecx = i
    imull %eax, %ecx          # ecx = i * j（旧值）
    testl %ecx, %ecx          # 测试 i*j
    setg %cl                  # cl = (i*j > 0)
    
    addl $1, %edx             # edx = i + 1
    movl %edx, 48(%esp)       # i++
    subl $1, %eax             # eax = j - 1
    movl %eax, 52(%esp)       # j--
    
    testb %cl, %cl            # 测试 cl
    jne .L6                   # 若旧 i*j > 0，继续循环
    
    movl 48(%esp), %eax       # eax = 新 i
    cmpl $4, %eax             
    jle .L6                   # 若新 i ≤ 4，继续循环
```

循环继续条件：
1. 用**旧值**计算 `i*j > 0`，若成立则继续；
2. 否则用**新 i**判断 `i ≤ 4`，若成立则继续。

C 语言中 `i++ * j--` 先取 i、j 的旧值做乘法，再执行自增/自减，正好对应汇编语义。后续 `i <= 4` 使用的是自增后的新 i。

综合得：`i++ * j-- > 0 || i <= 4`。

---

### (5) `*p + k - i*j`

汇编对应段（循环结束后的返回部分）：

```asm
    movl 60(%esp), %eax       # eax = p
    movl (%eax), %eax         # eax = *p
    movl %eax, %edx           # edx = *p
    addl 56(%esp), %edx       # edx = *p + k
    movl 48(%esp), %ecx       # ecx = i
    movl 52(%esp), %eax       # eax = j
    imull %ecx, %eax          # eax = i * j
    movl %edx, %ecx           # ecx = *p + k
    subl %eax, %ecx           # ecx = *p + k - i*j
    movl %ecx, %eax           # eax = 返回值
    leave
    ret
```

`eax` 最终值为 `*p + k - i*j`，作为 `main` 函数的返回值。

## 完整程序

```c
#include <stdio.h>
int main()
{
    int i,j,k=0,loop;
    int* p;
    int arr[3][2]={{9,5},{31,22},{6,18}};
    scanf("%d,%d",&i,&j);
    p=arr+2;
    do
    {
        if(i*j - *p > 6)
            k+=*p;
        else
            k+=*(p-4)>2 ? k-2*i : i+j;
    }while(i++ * j-- > 0 || i <= 4);
    return *p + k - i*j;
}
```
