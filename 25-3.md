# 程序填空题（第一空 4 分，其余两空每空 3 分，共 10 分）

## C 语言程序：

```c
    struct s *rchild;

    return (1) ;

int f2(int m,int n)
{
    return m*m+n*n;
}

int main()
{
    int i,j,k;
    root=(struct s*)malloc(sizeof(struct s));
    root->value=2025;
    root->lchild=NULL;
    root->rchild=NULL;
    ps=root;
    scanf("%d,%d\n",&i,&j);
    while(f2(i,j)!=0)
    {
        ps1=(struct s*)malloc(sizeof(struct s));
        ps1->lchild=NULL;
        ps1->rchild=NULL;ps1->value=i;
        ps2=(struct s*)malloc(sizeof(struct s));
        ps2->lchild=NULL;
        ps2->rchild=NULL;
        ps->lchild=ps1;
        ps->rchild=ps2; ps1->value=j;
        if( (2) )
            ps=ps->lchild;
        else
            ps=ps->rchild;
    }
    return (3) ;
}
```

## 答案解析

### 结构体成员偏移分析

从 `main` 中对 `root` 的初始化操作可以确定结构体 `s` 的成员偏移：

| 成员 | 偏移 |
|------|------|
| `value` | 0 |
| `lchild` | 4 |
| `rchild` | 8 |

### 空 (1)：`f1` 函数的返回值

分析 `f1` 汇编代码中 `.L2` 标号后的核心逻辑：

```asm
movl 8(%ebp), %eax       # eax = ps3
movl (%eax), %ebx        # ebx = ps3->value          -- ① 取出 value
movl 8(%ebp), %eax
movl 8(%eax), %eax       # eax = ps3->rchild
movl %eax, (%esp)
call f1                  # 递归调用 f1(ps3->rchild)
movl %eax, %edx          # edx = f1(ps3->rchild)
movl %edx, %eax
addl %eax, %eax          # eax = 2 * f1(ps3->rchild)
leal (%eax,%edx), %esi   # esi = 2*f1(rchild) + f1(rchild) = 3 * f1(ps3->rchild)  -- ②
movl 8(%ebp), %eax
movl 4(%eax), %eax       # eax = ps3->lchild
movl %eax, (%esp)
call f1                  # 递归调用 f1(ps3->lchild)
addl %eax, %eax          # eax = 2 * f1(ps3->lchild)   -- ③
addl %esi, %eax          # eax = 2*f1(lchild) + 3*f1(rchild)
addl %ebx, %eax          # eax = value + 2*f1(lchild) + 3*f1(rchild)  -- 最终结果
```

**答案 (1)：** `ps3->value + 2 * f1(ps3->lchild) + 3 * f1(ps3->rchild)`

### 空 (2)：`if` 条件判断

分析 `main` 中循环体内部的条件判断部分：

```asm
movl 24(%esp), %edx      # edx = i
movl 28(%esp), %eax      # eax = j
movl %edx, 4(%esp)       # 第二个参数 = i
movl %eax, (%esp)        # 第一个参数 = j
call f2                  # 调用 f2(j, i) = j² + i²
movl 24(%esp), %edx      # edx = i
movl %eax, 4(%esp)       # 第二个参数 = f2(j, i)
movl %edx, (%esp)        # 第一个参数 = i
call f2                  # 调用 f2(i, f2(j, i)) = i² + (i²+j²)²
cmpl $2025, %eax         # 比较结果与 2025
jle .L7                  # 若 <= 2025，跳转到 .L7（else 分支）
# 若不跳转（即 > 2025），继续执行 if 分支：
movl ps, %eax
movl 4(%eax), %eax       # ps = ps->lchild
movl %eax, ps
jmp .L6
.L7:                     # else 分支：
movl ps, %eax
movl 8(%eax), %eax       # ps = ps->rchild
movl %eax, ps
```

控制流分析：
- `jle .L7`：当 `f2(i, f2(j, i)) <= 2025` 时跳转到 `.L7`（else 分支）
- 不跳转时（即 `f2(i, f2(j, i)) > 2025`）执行 if 分支（`ps = ps->lchild`）

**答案 (2)：** `f2(i, f2(j, i)) > 2025`

### 空 (3)：`main` 函数的返回值

分析 `main` 结尾部分（循环结束后）：

```asm
movl root, %eax
movl 4(%eax), %eax       # eax = root->lchild
movl %eax, (%esp)
call f1                  # 调用 f1(root->lchild)
movl %eax, %ebx          # ebx = f1(root->lchild)
movl root, %eax
movl %eax, (%esp)
call f1                  # 调用 f1(root)
movl %ebx, 4(%esp)       # 第二个参数 = f1(root->lchild)
movl %eax, (%esp)        # 第一个参数 = f1(root)
call f2                  # 调用 f2(f1(root), f1(root->lchild))
                         # 返回值在 %eax 中，作为 main 的返回值
```

**答案 (3)：** `f2(f1(root), f1(root->lchild))`
